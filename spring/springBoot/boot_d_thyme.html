# thymeleafжЁЎжқҝеј•ж“Һзҡ„дҪҝз”Ё

вҖӢ еңЁж—©жңҹејҖеҸ‘зҡ„ж—¶еҖҷпјҢжҲ‘д»¬е®ҢжҲҗзҡ„йғҪжҳҜйқҷжҖҒйЎөйқўд№ҹе°ұжҳҜhtmlйЎөйқўпјҢйҡҸзқҖж—¶й—ҙиҪҙзҡ„еҸ‘еұ•пјҢж…ўж…ўзҡ„еј•е…ҘдәҶjspйЎөйқўпјҢеҪ“еңЁеҗҺз«ҜжңҚеҠЎжҹҘиҜўеҲ°ж•°жҚ®д№ӢеҗҺеҸҜд»ҘиҪ¬еҸ‘еҲ°jspйЎөйқўпјҢеҸҜд»ҘиҪ»жқҫзҡ„дҪҝз”ЁjspйЎөйқўжқҘе®һзҺ°ж•°жҚ®зҡ„жҳҫзӨәеҸҠдәӨдә’пјҢjspжңүйқһеёёејәеӨ§зҡ„еҠҹиғҪпјҢдҪҶжҳҜпјҢеңЁдҪҝз”Ёspringbootзҡ„ж—¶еҖҷпјҢж•ҙдёӘйЎ№зӣ®жҳҜд»ҘjarеҢ…зҡ„ж–№ејҸиҝҗиЎҢиҖҢдёҚжҳҜwarеҢ…пјҢиҖҢдё”иҝҳеөҢе…ҘдәҶtomcatе®№еҷЁпјҢеӣ жӯӨпјҢеңЁй»ҳи®Өжғ…еҶөдёӢжҳҜдёҚж”ҜжҢҒjspйЎөйқўзҡ„гҖӮеҰӮжһңзӣҙжҺҘд»ҘзәҜйқҷжҖҒйЎөйқўзҡ„ж–№ејҸдјҡз»ҷжҲ‘д»¬зҡ„ејҖеҸ‘еёҰжқҘеҫҲеӨ§зҡ„йә»зғҰпјҢspringbootжҺЁиҚҗдҪҝз”ЁжЁЎжқҝеј•ж“ҺгҖӮ

вҖӢ жЁЎжқҝеј•ж“ҺжңүеҫҲеӨҡз§ҚпјҢjsp,freemarker,thymeleafпјҢжЁЎжқҝеј•ж“Һзҡ„дҪңз”Ёе°ұжҳҜжҲ‘д»¬жқҘеҶҷдёҖдёӘйЎөйқўжЁЎжқҝпјҢжҜ”еҰӮжңүдәӣеҖје‘ўпјҢжҳҜеҠЁжҖҒзҡ„пјҢжҲ‘д»¬еҶҷдёҖдәӣиЎЁиҫҫејҸгҖӮиҖҢиҝҷдәӣеҖјпјҢд»Һе“ӘжқҘе‘ўпјҢжҲ‘д»¬жқҘз»„иЈ…дёҖдәӣж•°жҚ®пјҢжҲ‘д»¬жҠҠиҝҷдәӣж•°жҚ®жүҫеҲ°гҖӮз„¶еҗҺжҠҠиҝҷдёӘжЁЎжқҝе’ҢиҝҷдёӘж•°жҚ®дәӨз»ҷжҲ‘д»¬жЁЎжқҝеј•ж“ҺпјҢжЁЎжқҝеј•ж“ҺжҢүз…§жҲ‘д»¬иҝҷдёӘж•°жҚ®её®дҪ жҠҠиҝҷиЎЁиҫҫејҸи§ЈжһҗгҖҒеЎ«е……еҲ°жҲ‘д»¬жҢҮе®ҡзҡ„дҪҚзҪ®пјҢз„¶еҗҺжҠҠиҝҷдёӘж•°жҚ®жңҖз»Ҳз”ҹжҲҗдёҖдёӘжҲ‘д»¬жғіиҰҒзҡ„еҶ…е®№з»ҷжҲ‘д»¬еҶҷеҮәеҺ»пјҢиҝҷе°ұжҳҜжҲ‘д»¬иҝҷдёӘжЁЎжқҝеј•ж“ҺпјҢдёҚз®ЎжҳҜjspиҝҳжҳҜе…¶д»–жЁЎжқҝеј•ж“ҺпјҢйғҪжҳҜиҝҷдёӘжҖқжғігҖӮеҸӘдёҚиҝҮдёҚеҗҢзҡ„жЁЎжқҝеј•ж“ҺиҜӯжі•дёҚеҗҢиҖҢе·І

# 1гҖҒthзҡ„еёёз”ЁеұһжҖ§еҖј

вҖӢ дёҖгҖҒth:text пјҡи®ҫзҪ®еҪ“еүҚе…ғзҙ зҡ„ж–Үжң¬еҶ…е®№пјҢзӣёеҗҢеҠҹиғҪзҡ„иҝҳжңүth:utextпјҢдёӨиҖ…зҡ„еҢәеҲ«еңЁдәҺеүҚиҖ…дёҚдјҡиҪ¬д№үhtmlж ҮзӯҫпјҢеҗҺиҖ…дјҡгҖӮдјҳе…Ҳзә§дёҚй«ҳпјҡorder=7

вҖӢ дәҢгҖҒth:valueпјҡи®ҫзҪ®еҪ“еүҚе…ғзҙ зҡ„valueеҖјпјҢзұ»дјјдҝ®ж”№жҢҮе®ҡеұһжҖ§зҡ„иҝҳжңүth:srcпјҢth:hrefгҖӮдјҳе…Ҳзә§дёҚй«ҳпјҡorder=6

вҖӢ дёүгҖҒth:eachпјҡйҒҚеҺҶеҫӘзҺҜе…ғзҙ пјҢе’Ңth:textжҲ–th:valueдёҖиө·дҪҝз”ЁгҖӮжіЁж„ҸиҜҘеұһжҖ§дҝ®йҘ°зҡ„ж ҮзӯҫдҪҚзҪ®пјҢиҜҰз»ҶеҫҖеҗҺзңӢгҖӮдјҳе…Ҳзә§еҫҲй«ҳпјҡorder=2

вҖӢ еӣӣгҖҒth:ifпјҡжқЎд»¶еҲӨж–ӯпјҢзұ»дјјзҡ„иҝҳжңүth:unlessпјҢth:switchпјҢth:caseгҖӮдјҳе…Ҳзә§иҫғй«ҳпјҡorder=3

вҖӢ дә”гҖҒth:insertпјҡд»Јз Ғеқ—еј•е…ҘпјҢзұ»дјјзҡ„иҝҳжңүth:replaceпјҢth:includeпјҢдёүиҖ…зҡ„еҢәеҲ«иҫғеӨ§пјҢиӢҘдҪҝз”ЁдёҚжҒ°еҪ“дјҡз ҙеқҸhtmlз»“жһ„пјҢеёёз”ЁдәҺе…¬е…ұд»Јз Ғеқ—жҸҗеҸ–зҡ„еңәжҷҜгҖӮдјҳе…Ҳзә§жңҖй«ҳпјҡorder=1

вҖӢ е…ӯгҖҒth:fragmentпјҡе®ҡд№үд»Јз Ғеқ—пјҢж–№дҫҝиў«th:insertеј•з”ЁгҖӮдјҳе…Ҳзә§жңҖдҪҺпјҡorder=8

вҖӢ дёғгҖҒth:objectпјҡеЈ°жҳҺеҸҳйҮҸпјҢдёҖиҲ¬е’Ң*{}дёҖиө·й…ҚеҗҲдҪҝз”ЁпјҢиҫҫеҲ°еҒ·жҮ’зҡ„ж•ҲжһңгҖӮдјҳе…Ҳзә§дёҖиҲ¬пјҡorder=4

вҖӢ е…«гҖҒth:attrпјҡдҝ®ж”№д»»ж„ҸеұһжҖ§пјҢе®һйҷ…ејҖеҸ‘дёӯз”Ёзҡ„иҫғе°‘пјҢеӣ дёәжңүдё°еҜҢзҡ„е…¶д»–thеұһжҖ§её®еҝҷпјҢзұ»дјјзҡ„иҝҳжңүth:attrappendпјҢth:attrprependгҖӮдјҳе…Ҳзә§дёҖиҲ¬пјҡorder=5

package com.osvue.controller;

        map.put("thText","th:textи®ҫзҪ®ж–Үжң¬еҶ…е®№ <b>еҠ зІ—</b>");
        map.put("thUText","th:utext и®ҫзҪ®ж–Үжң¬еҶ…е®№ <b>еҠ зІ—</b>");

        map.put("thObject",new Person("zhangsan",12,"з”·"));
        return "thymeleaf";
    }
}

<!--

	@Override
	public Locale resolveLocale(HttpServletRequest request) {
		Locale defaultLocale = getDefaultLocale();
		if (defaultLocale != null && request.getHeader("Accept-Language") == null) {
			return defaultLocale;
		}
		Locale requestLocale = request.getLocale();
		List<Locale> supportedLocales = getSupportedLocales();
		if (supportedLocales.isEmpty() || supportedLocales.contains(requestLocale)) {
			return requestLocale;
		}
		Locale supportedLocale = findSupportedLocale(request, supportedLocales);
		if (supportedLocale != null) {
			return supportedLocale;
		}
